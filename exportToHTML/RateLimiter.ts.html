/**
 * Implementación simple de Rate Limiting para la capa de Infraestructura. 
 * Protege contra ataques de fuerza bruta y tráfico excesivo. 
 */
class RateLimiter {
  private requests: Map<string, number[]> = new Map();
  private readonly LIMIT = 10; // Máximo 10 peticiones
  private readonly WINDOW_MS = 60000; // Por minuto (60 segundos)

  /**
   * Verifica si una acción está permitida para una clave dada (ej: IP o UserID). 
   */
  isAllowed(key: string): boolean {
    const now = Date.now();
    const userRequests = this.requests.get(key) || [];

    // Filtrar peticiones fuera de la ventana de tiempo
    const recentRequests = userRequests.filter(timestamp => now - timestamp < this.WINDOW_MS);

    if (recentRequests.length >= this.LIMIT) {
      console.warn(`[RateLimit] Bloqueado tráfico excesivo para: ${key}`);
      return false;
    }

    recentRequests.push(now);
    this.requests.set(key, recentRequests);
    return true;
  }
}

export const globalRateLimiter = new RateLimiter();
