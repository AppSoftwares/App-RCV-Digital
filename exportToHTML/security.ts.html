/**
 * Utilidades de seguridad para sanitización y validación. 
 */

/**
 * Sanitiza la entrada para evitar inyección NoSQL y XSS. 
 * Fuerza el tipo a string y elimina operadores de MongoDB ($). 
 */
export const sanitizeInput = (input: unknown): string => {
  if (typeof input !== 'string') return "";

  return input
    .replace(/\$/g, "") // Elimina el signo de dólar (operador NoSQL)
    .replace(/[<>'";\\]/g, "") // XSS básico
    .trim();
};

/**
 * Normaliza el email para consistencia y evitar bypass. 
 */
export const normalizeEmail = (email: unknown): string => {
  if (typeof email !== 'string') return "";
  return email.toLowerCase().trim();
};

/**
 * Validación estricta de entrada de login. 
 * Verifica tipo, longitud y charset permitido. 
 */
export const validateAuthInput = (input: unknown): { isValid: boolean; value: string } => {
  if (typeof input !== 'string') return { isValid: false, value: "" };

  const value = input.trim();

  // Regla de longitud: evitar ataques de desbordamiento de buffer o DoS
  if (value.length < 3 || value.length > 100) return { isValid: false, value: "" };

  // Charset alfanumérico + símbolos básicos seguros
  const safeCharset = /^[a-zA-Z0-9@\._-]+$/;
  if (!safeCharset.test(value)) return { isValid: false, value: "" };

  return { isValid: true, value };
};

export const validateEmail = (email: string): boolean => {
  const re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return re.test(email);
};

export const validateRIF = (rif: string): boolean => {
  const re = /^[JGVEn0-9-]+$/i;
  return re.test(rif);
};
