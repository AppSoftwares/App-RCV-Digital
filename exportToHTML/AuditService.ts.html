import { UserRole } from '../../types';

export interface AuditEntry {
  id: string;
  timestamp: string;
  userId?: string;
  userRole?: UserRole;
  action: string;
  category: 'security' | 'accounting' | 'admin';
  severity: 'low' | 'medium' | 'high' | 'critical';
  details: string;
  ip?: string;
  metadata?: any;
}

export class AuditService {
  private logs: AuditEntry[] = [];

  /**
   * Registra un evento en el log de auditoría. 
   * En producción, esto enviaría los datos a una tabla de auditoría en DB o a un SIEM (ej: Datadog, ELK). 
   */
  log(entry: Omit<AuditEntry, 'id' | 'timestamp'>): void {
    const newEntry: AuditEntry = {
      ...entry,
      id: `audit_${Date.now()}_${Math.random().toString(36).substr(2, 5)}`,
      timestamp: new Date().toISOString()
    };

    this.logs.unshift(newEntry); // Mantener los más recientes arriba

    // Monitoreo de Patrones Raros: Alerta inmediata si es crítico
    if (entry.severity === 'critical' || entry.severity === 'high') {
      console.error(`[SECURITY CRITICAL] ${entry.action}: ${entry.details}`);
      // Aquí se dispararía una notificación (Push, Email, Slack)
    } else {
      console.log(`[Audit] ${entry.category.toUpperCase()} - ${entry.action}`);
    }

    // Limitar logs en memoria (solo para propósitos de demo)
    if (this.logs.length > 100) this.logs.pop();
  }

  getLogs(category?: AuditEntry['category']): AuditEntry[] {
    if (category) return this.logs.filter(l => l.category === category);
    return this.logs;
  }

  /**
   * Detecta patrones raros comparando eventos recientes. 
   */
  detectAnomalies(): string[] {
    const anomalies: string[] = [];
    const recentSecurityLogs = this.logs.filter(l =>
      l.category === 'security' &&
      (Date.now() - new Date(l.timestamp).getTime()) < 3600000 // última hora
    );

    // Patrón 1: Múltiples fallos de login en corto tiempo (Fuerza Bruta)
    const loginFailures = recentSecurityLogs.filter(l => l.action === 'LOGIN_FAILED');
    if (loginFailures.length > 5) {
      anomalies.push("Detección de posible ataque de fuerza bruta (Múltiples fallos de login)");
    }

    // Patrón 2: Acceso desde IPs inusuales o múltiples IPs para un mismo usuario
    // (Simulado)

    return anomalies;
  }
}

export const auditService = new AuditService();
