import React from 'react';
import { Home, ShieldCheck, Wallet, User } from 'lucide-react';

interface BottomNavProps {
  activeTab: 'home' | 'policies' | 'payments' | 'profile';
}

export const BottomNav: React.FC<BottomNavProps> = ({ activeTab }) => (
  <nav className="fixed bottom-0 left-0 w-full z-50 flex justify-around items-center px-4 py-3 bg-slate-950/50 backdrop-blur-md border-t border-white/5 md:hidden">
    <NavItem icon={Home} label="Inicio" active={activeTab === 'home'} />
    <NavItem icon={ShieldCheck} label="Pólizas" active={activeTab === 'policies'} />
    <NavItem icon={Wallet} label="Pagos" active={activeTab === 'payments'} />
    <NavItem icon={User} label="Perfil" active={activeTab === 'profile'} />
  </nav>
);

const NavItem = ({ icon: Icon, label, active }: { icon: any, label: string, active: boolean }) => (
  <button className={`flex flex-col items-center justify-center gap-1 transition-all ${active ? 'text-brand-primary' : 'text-slate-500'}`}>
    <div className={`p-1 rounded-xl ${active ? 'bg-brand-primary/10' : ''}`}>
      <Icon className="w-6 h-6" />
    </div>
    <span className="text-[10px] font-medium">{label}</span>
  </button>
);
