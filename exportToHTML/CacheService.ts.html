/**
 * Capa de Caché para optimización de rendimiento (Regla 80/20). 
 * Almacena en memoria los datos accedidos frecuentemente para reducir carga en BD y latencia. 
 */

interface CacheItem<T> {
  data: T;
  expiresAt: number;
}

export class CacheService {
  private cache: Map<string, CacheItem<any>> = new Map();
  private readonly DEFAULT_TTL = 300000; // 5 minutos por defecto

  /**
   * Obtiene o establece un valor en el caché. 
   * @param key Identificador único del dato. 
   * @param fetcher Función para obtener el dato si no está en caché. 
   * @param ttl Tiempo de vida en milisegundos. 
   */
  async getOrSet<T>(key: string, fetcher: () => Promise<T>, ttl: number = this.DEFAULT_TTL): Promise<T> {
    const cached = this.cache.get(key);
    const now = Date.now();

    if (cached && cached.expiresAt > now) {
      console.log(`[Cache] HIT: ${key}`);
      return cached.data;
    }

    console.log(`[Cache] MISS: ${key}. Consultando fuente original...`);
    const data = await fetcher();
    this.cache.set(key, { data, expiresAt: now + ttl });

    return data;
  }

  /**
   * Invalida una entrada específica (ej: al actualizar un usuario). 
   */
  invalidate(key: string): void {
    this.cache.delete(key);
  }

  /**
   * Limpia todo el caché. 
   */
  clear(): void {
    this.cache.clear();
  }
}

export const globalCache = new CacheService();
